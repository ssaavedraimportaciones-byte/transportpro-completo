Eres mi asistente de desarrollo para TransportPro, un SaaS de gestión
de flotas para empresas de transporte en Chile.

## CONTEXTO TÉCNICO

- App web HTML/JS vanilla + Supabase backend
- Archivo principal: index.html (login + todos los módulos)
- Supabase URL: https://wylmxqjpkdqxtqjkuqqn.supabase.co
- CRÍTICO: siempre usar `const sb = window.supabase.createClient(URL, KEY)`
  NUNCA usar `const supabase` (causa SyntaxError silencioso)
- RLS habilitada: cada empresa solo ve sus datos (filtrar por empresa_id)
- Formato moneda: toLocaleString('es-CL') — pesos chilenos sin decimales
- Fechas mostrar: DD/MM/YYYY — guardar: ISO 8601

## BASE DE DATOS ACTUAL

- empresas: id, nombre
- usuarios: id, empresa_id, rol
- vehiculos: id, empresa_id, patente, marca, modelo
- conductores: id, empresa_id, nombre, rut
- clientes: id, empresa_id, nombre, rut
- viajes: id, empresa_id, vehiculo_id, conductor_id, cliente_id,
          monto, descripcion, fecha, estado
- combustible: id, empresa_id, vehiculo_id, litros, costo, fecha
- neumaticos: id, empresa_id, vehiculo_id, descripcion, costo, fecha
- mantenimiento: id, empresa_id, vehiculo_id, descripcion, costo, fecha
- gastos: id, empresa_id, descripcion, monto, fecha, categoria
- facturas: id, empresa_id, viaje_id, numero_dte, pdf_url, estado

-- 4. Liquidaciones
CREATE TABLE IF NOT EXISTS liquidaciones (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  empresa_id UUID REFERENCES empresas(id),
  conductor_id UUID REFERENCES conductores(id),
  fecha_inicio DATE NOT NULL,
  fecha_fin DATE NOT NULL,
  total_bruto INTEGER DEFAULT 0,
  total_descuentos INTEGER DEFAULT 0,
  total_neto INTEGER DEFAULT 0,
  estado TEXT CHECK (estado IN (
    'borrador','emitida','pagada'
  )) DEFAULT 'borrador',
  observaciones TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS liquidacion_items (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  liquidacion_id UUID REFERENCES liquidaciones(id) ON DELETE CASCADE,
  tipo TEXT CHECK (tipo IN ('ingreso','descuento')),
  concepto TEXT NOT NULL,
  referencia_id UUID,
  monto INTEGER NOT NULL,
  fecha DATE
);

-- 5. RLS nuevas tablas
ALTER TABLE liquidaciones ENABLE ROW LEVEL SECURITY;
ALTER TABLE liquidacion_items ENABLE ROW LEVEL SECURITY;
ALTER TABLE tipos_gasto ENABLE ROW LEVEL SECURITY;

CREATE POLICY "empresa_liquidaciones" ON liquidaciones
  FOR ALL USING (empresa_id = (
    SELECT empresa_id FROM usuarios WHERE id = auth.uid()
  ));

CREATE POLICY "empresa_liquidacion_items" ON liquidacion_items
  FOR ALL USING (liquidacion_id IN (
    SELECT id FROM liquidaciones WHERE empresa_id = (
      SELECT empresa_id FROM usuarios WHERE id = auth.uid()
    )
  ));

CREATE POLICY "tipos_gasto_publicos" ON tipos_gasto
  FOR SELECT USING (
    empresa_id IS NULL OR empresa_id = (
      SELECT empresa_id FROM usuarios WHERE id = auth.uid()
    )
  );

-- 6. Vista facturas vencidas
CREATE OR REPLACE VIEW facturas_vencidas AS
SELECT f.*, c.nombre as cliente_nombre, c.rut as cliente_rut,
       CURRENT_DATE - f.fecha_vencimiento as dias_mora
FROM facturas f
JOIN clientes c ON c.id = f.viaje_id
WHERE f.estado_cobro = 'pendiente'
  AND f.fecha_vencimiento < CURRENT_DATE;

function exportarExcel(datos, nombreArchivo, nombreHoja) {
  const ws = XLSX.utils.json_to_sheet(datos);
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws, nombreHoja);
  const mes = new Date().toISOString().slice(0,7);
  XLSX.writeFile(wb, nombreArchivo + '_' + mes + '.xlsx');
}
